	enable = {
		USA = { has_war_with = SPR }
		ROOT = { has_global_flag = Spanish_American_War_Occured }
	}
	abort = {
		USA = { NOT = { has_war_with = SPR } }
	}

	ai_strategy = {
		type = invade
		id = SPR
		value = 200
	}

	ai_strategy = {
		type = antagonize
		id = SPR
		value = 200
	}
}
